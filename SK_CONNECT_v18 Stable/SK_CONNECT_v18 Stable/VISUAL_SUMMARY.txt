━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

event_registrations (Main Table)
├─ id: Unique identifier
├─ event_id: Link to event
├─ full_name: Participant name
├─ barangay: Location
├─ contact: Phone number
├─ email: Email address
├─ registration_token: Unique QR token
├─ registered_at: Registration timestamp
├─ attended: true/false
├─ attended_at: Attendance timestamp
├─ marked_by: Admin who marked
└─ created_at: Record creation time

attendances (Generic Records)
├─ id: Unique identifier
├─ event_id: Link to event
├─ registration_id: Link to registration
├─ user_id: Link to user
├─ name: Person name
├─ email: Email address
├─ phone: Phone number
├─ barangay: Location
├─ status: Present/Absent/Pending
├─ marked_at: Marked timestamp
└─ created_at: Record creation time

🔧 API ENDPOINTS AT A GLANCE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

GET /api/attendance/{eventId}
  ├─ Returns: All registrations for event
  ├─ Plus: Statistics (total, attended, pending)
  └─ Used by: Attendance modal

POST /api/attendance/mark
  ├─ Input: QR token
  ├─ Action: Mark attendance
  └─ Used by: QR scanner

PATCH /api/attendance/{eventId}/{registrationId}
  ├─ Input: status (Present/Absent/Pending)
  ├─ Action: Update status
  └─ Used by: Checkbox toggle

GET /api/attendance/{eventId}/export
  ├─ Returns: CSV file
  ├─ Contains: All attendance data
  └─ Used by: Export button

POST /api/event/{eventId}/generate-token
  ├─ Returns: Token + QR URL
  ├─ Action: Generate unique token
  └─ Used by: QR code generation
